#include <stdio.h>
#include <memory.h>
#define size 101
#define inf 32767
typedef struct
{
    int n,e;
    int edge[size][size];
}mg;

void init_mg(mg &g)
{
    g.n=g.e=0;
    for (int i = 1; i < size; i++) {
        for (int j = 0; j < size; j++) {
            g.edge[i][j]=inf;
        }

    }
}
void input(mg &g)
{
    freopen("in.txt","r",stdin);
    scanf("%d%d",&g.n,&g.e);
    for (int i = 1; i <= g.e ; i++) {
        int v1,v2,w;
        scanf("%d%d%d",&v1,&v2,&w);
        g.edge[v1][v2]=g.edge[v2][v1]=w;
    }
    fclose(stdin);
}
typedef struct {
    int vs,ve,w;
}eg;
void sort(eg g[],int e)
{
    for (int i = 1; i < e; i++) {
        for (int j = i+1; j <= e; j++) {
            if(g[i].w>g[j].w)
            {
                eg t=g[i];
                g[i]=g[j];
                g[j]=t;
            }
        }

    }
}
void kruskal(mg g)
{
    eg eg[size];
    int ei=0;
    for(int i=1;i<=g.n;i++)
        for(int j=1;j<i;j++)
            if(g.edge[i][j]<inf)
            {
                ei++;
                eg[ei].vs=i;
                eg[ei].ve=j;
                eg[ei].w=g.edge[i][j];
            }
            sort(eg,ei);
    int sn[size]={0};
    for(int i=1;i<g.n;i++)
        sn[i]=i;
    int k=0;
    ei=1;
    while(1)
    {
        int vs,ve,w,sn1,sn2;
        vs=eg[ei].vs;
        ve=eg[ei].ve;
        w=eg[ei].w;
        sn1=sn[vs];
        sn2=sn[ve];
        if(sn1!=sn2)
        {
            printf("(%d,%d)(%d),",ve,vs,w);
            k++;
            if(k==g.n-1) break;
            for(int i=1;i<=g.n;i++)
            {
                if(sn[i]==sn2) sn[i]=sn1;
            }
        }
        ei++;
    }
}
int Min(eg eg[],int n)
{
	int min=inf,mini=0;
	for(int i=1;i<n;i++)
		if(min>eg[i].w)
		{
			min=eg[i].w;
			mini=i;
		}
	return mini;
}
void pulimu(mg g)
{
	int b[size]={0};
	b[1]=1;
	int k=1;
	while(k<g.n)
	{
		eg eg[size];
		int ei=0;
		for(int i=1;i<=g.n;i++)
			for(int j=1;j<=g.n;j++)
				if(b[i]&&!b[j]&&g.edge[i][j]<inf)
				{
					ei++;
					eg[ei].vs=i;eg[ei].ve=j;
					eg[ei].w=g.edge[i][j];
				}
				int m=Min(eg,ei);
				printf("(%d,%d)(%d),",eg[m].vs,eg[m].ve,eg[m].w);
				b[eg[m].ve]=1;
				k++;
	}
}
int serchmin(int lamd[],bool blamd[],int n)
{
	int k=1;
	int min=inf;
	for(int i=1;i<=n;i++)
	{
		if(min>lamd[i]&&!blamd[i])
		{
			k=i;
			min=lamd[i];
		}
	}
	return k;
}
void dikesiche(mg g,int vstart)
{
	int lamd[size],dis[size][size];
	bool blamd[size];
    memset(lamd,inf,100);
	memset(blamd,false,100);
	printf("%d",g.n);
	for(int i=1;i<=g.n;i++)
		for(int j=1;j<=g.n;i++)
			dis[i][j]=g.edge[i][j];
	blamd[vstart]=true;
	for(int i=1;i<=g.n;i++)
	{
		lamd[i]=dis[vstart][i];
	}
	for(int i=2;i<=g.n;i++)
	{
		int k=serchmin(lamd,blamd,g.n);
		blamd[k]=true;
		printf("%d(%d),",k,lamd[k]);
		for(int j=1;j<=g.n;j++)
		{
			if(lamd[j]>lamd[k]+dis[k][j])
				lamd[j]=lamd[k]+dis[k][j];
		}
	}
}
int main()
{
    mg g;
    init_mg(g);
    input(g);
    printf("%d",g.n);
    kruskal(g);
    pulimu(g);
	dikesiche(g,1);
    return 0;
}
