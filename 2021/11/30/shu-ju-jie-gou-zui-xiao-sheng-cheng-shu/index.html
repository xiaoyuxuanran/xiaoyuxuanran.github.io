数据结构最小生成树

发表于 2021-11-30 分类于 Markdown 
#include <stdio.h>
#define size 101
#define inf 32767
typedef struct
{
    int n,e;
    int edge[size][size];
}mg;

void init_mg(mg &g)
{
    g.n=g.e=0;
    for (int i = 1; i < size; i++) {
        for (int j = 0; j < size; j++) {
            g.edge[i][j]=inf;
        }

    }
}
void input(mg &g)
{
    freopen("in.txt","r",stdin);
    scanf("%d%d",&g.n,&g.e);
    for (int i = 1; i <= g.e ; i++) {
        int v1,v2,w;
        scanf("%d%d%d",&v1,&v2,&w);
        g.edge[v1][v2]=g.edge[v2][v1]=w;
    }
    fclose(stdin);
}
typedef struct {
    int vs,ve,w;
}eg;
void sort(eg g[],int e)
{
    for (int i = 1; i < e; i++) {
        for (int j = i+1; j <= e; j++) {
            if(g[i].w>g[j].w)
            {
                eg t=g[i];
                g[i]=g[j];
                g[j]=t;
            }
        }

    }
}
void kruskal(mg g)
{
    eg eg[size];
    int ei=0;
    for(int i=1;i<=g.n;i++)
        for(int j=1;j<i;j++)
            if(g.edge[i][j]<inf)
            {
                ei++;
                eg[ei].vs=i;
                eg[ei].ve=j;
                eg[ei].w=g.edge[i][j];
            }
            sort(eg,ei);
    int sn[size]={0};
    for(int i=1;i<g.n;i++)
        sn[i]=i;
    int k=0;
    ei=1;
    while(1)
    {
        int vs,ve,w,sn1,sn2;
        vs=eg[ei].vs;
        ve=eg[ei].ve;
        w=eg[ei].w;
        sn1=sn[vs];
        sn2=sn[ve];
        if(sn1!=sn2)
        {
            printf("(%d,%d)(%d),",vs,ve,w);
            k++;
            if(k==g.n-1) break;
            for(int i=1;i<=g.n;i++)
            {
                if(sn[i]==sn2) sn[i]=sn1;
            }
        }
        ei++;
    }
}
int main()
{
    mg g;
    init_mg(g);
    input(g);
    kruskal(g);
    return 0;
}
